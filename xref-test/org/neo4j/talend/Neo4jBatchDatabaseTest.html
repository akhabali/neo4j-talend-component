18  
19          // populate the index
20          for (int i = 0; i < 100; i++) {
21              batchDb.indexNodeInBatchIndex(INDEX_NAME, Long.valueOf(i), Long.valueOf(1000 + i));
22          }
23  
24          // flush it

27          // find node into index & make assert
